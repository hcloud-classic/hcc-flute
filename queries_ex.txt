//// node
// list_node : Get (read) node list
http://localhost:7000/graphql?query={list_node{uuid,mac_addr,ipmi_ip,status,cpu,memory,detail,created_at}}

// node : Get a node by uuid
http://localhost:7000/graphql?query={node(uuid:"d4f3a900-b674-11e8-906e-000ffee02d5c"){uuid,mac_addr,ipmi_ip,status,cpu,memory,detail,created_at}}

// create : Create new node
http://localhost:7000/graphql?query=mutation+_{create_node(ipmi_ip:"172.31.0.1",detail:"Compute1"){uuid,mac_addr,ipmi_ip,status,cpu,memory,detail,created_at}}

// update_all_nodes : Update all infos of all nodes
http://localhost:7000/graphql?query=mutation+_{update_all_nodes(){uuid,mac_addr,ipmi_ip,status,cpu,memory,detail,created_at}}

// update_status_node : Update status of the node
http://localhost:7000/graphql?query=mutation+_{update_status_node(uuid:"d4f3a900-b674-11e8-906e-000ffee02d5c"){status}}

// update_status_nodes : Update status of all nodes
http://localhost:7000/graphql?query=mutation+_{update_status_nodes(){status}}
